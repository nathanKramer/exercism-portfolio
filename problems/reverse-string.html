Nim

proc reverse*(s: string): string =
  if s == "":
    return s

  var aux = s
  for i in 0 .. high(s) div 2:
    swap(aux[i], aux[high(aux) - i])
  aux

(defn reverse-string [s]
  (let [reversed-chars (into () s)] ;; https://clojuredocs.org/clojure.core/into
  (apply str reversed-chars)))
