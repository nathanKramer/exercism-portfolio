module Etl exposing (transform)

import Dict exposing (Dict)


transform : Dict Int (List String) -> Dict String Int
transform input =
    Dict.toList input
        |> List.concatMap (\( score, letters ) -> List.map (\l -> ( String.toLower l, score )) letters)
        |> Dict.fromList
